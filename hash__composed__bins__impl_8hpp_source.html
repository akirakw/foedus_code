268 inline ErrorCode ComposedBinsBuffer::next_bin() {
269  if (LIKELY(cursor_bin_ + 1U < cursor_bin_count_)) {
270  ++cursor_bin_;
271  return kErrorCodeOk;
272  } else if (LIKELY(cursor_buffer_ + 1U < buffer_count_)) {
273  ++cursor_buffer_;
274  cursor_bin_ = 0;
275  cursor_bin_count_ = buffer_[cursor_buffer_].bin_count_;
276  return kErrorCodeOk;
277  } else {
278  return next_pages();
279  }
280 }
