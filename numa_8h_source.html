96 static inline int nodemask_isset_compat(const nodemask_t *mask, int node)
97 {
98  if ((unsigned)node >= NUMA_NUM_NODES)
99  return 0;
100  if (mask->n[node / (8*sizeof(unsigned long))] &
101  (1UL<<(node%(8*sizeof(unsigned long)))))
102  return 1;
103  return 0;
104 }
