42 ErrorStack SnapshotMetadata::load(tinyxml2::XMLElement* element) {
43  clear();
44  EXTERNALIZE_LOAD_ELEMENT(element, id_);
45  EXTERNALIZE_LOAD_ELEMENT(element, base_epoch_);
46  EXTERNALIZE_LOAD_ELEMENT(element, valid_until_epoch_);
47  EXTERNALIZE_LOAD_ELEMENT(element, largest_storage_id_);
48 
49  uint64_t memory_size
50  = static_cast<uint64_t>(largest_storage_id_ + 1) * soc::GlobalMemoryAnchors::kStorageMemorySize;
51  storage_control_blocks_memory_.alloc(
52  memory_size,
53  1 << 12,
54  memory::AlignedMemory::kNumaAllocOnnode,
55  0);
56  storage_control_blocks_ = reinterpret_cast<storage::StorageControlBlock*>(
57  storage_control_blocks_memory_.get_block());
58  std::memset(storage_control_blocks_, 0, storage_control_blocks_memory_.get_size());
59 
60  // <storages>
61  tinyxml2::XMLElement* storages = element->FirstChildElement(kStoragesTagName);
62  if (!storages) {
63  // <storages> tag is missing. treat it as no storages. but weird!
64  LOG(ERROR) << "WAIT, the snapshot metadata file doesn't have " << kStoragesTagName
65  << " element? that's weird. It'll be treated as empty, but this shouldn't happen!";
66  } else {
67  CHECK_ERROR(storage::MetadataSerializer::load_all_storages_from_xml(
68  largest_storage_id_,
69  storages,
70  storage_control_blocks_));
71  }
72  // </storages>
73  return kRetOk;
74 }

76 ErrorStack SnapshotMetadata::save(tinyxml2::XMLElement* element) const {

78 
79  EXTERNALIZE_SAVE_ELEMENT(element, id_, "Unique ID of this snapshot");

81  "This snapshot was taken based on another snapshot that is valid_until this epoch."
82  " If this is the first snapshot, this value is 0.");
83  EXTERNALIZE_SAVE_ELEMENT(element, valid_until_epoch_,
84  "This snapshot contains all the logs until this epoch.");
85  EXTERNALIZE_SAVE_ELEMENT(element, largest_storage_id_,
86  "The largest StorageId we so far observed.");
87 
88  // <storages>

90  CHECK_OUTOFMEMORY(storages);

93  CHECK_ERROR(storage::MetadataSerializer::save_all_storages_to_xml(
94  largest_storage_id_,
95  storages,
96  storage_control_blocks_));
97  // </storages>
